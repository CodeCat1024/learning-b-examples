-- 查看数据库
show databases;

-- 创建数据库
create database testdatabase;

-- 创建数据库（不存在的话）
create database if not exists testdatabase;

-- 创建数据库（使用utf8mb4字符集）
create database if not exists default character set utf8mb4;

-- 使用数据库
use testdatabase;

-- 删除数据库
drop database testdatabase;

-- 删除数据库（如果存在的话）
drop database if exists testdatabase;
